    **2. `DeviceWorker` class — per-device poll loop (D-01, D-05, D-06, D-07, D-08, D-09, D-10, D-11):**
    ```python
    MAX_BACKOFF = 300  # 5 minutes, per D-06
    BACKOFF_FACTOR = 2

    class DeviceWorker:
        def __init__(self, config: DeviceConfig, parser: ParameterParser, client: Solarman, is_solar: bool):
            self.config = config
            self.parser = parser
            self.client = client
            self.logger = get_device_logger(config.name)
            self.health = DeviceHealth(is_solar)

            # Scheduling
            self._cycle_count: int = 0
            self._poll_interval: int = config.poll_interval

            # Backoff
            self._consecutive_failures: int = 0
            self._backoff_interval: int = config.poll_interval

            # Anti-overlap
            self._polling_in_progress: bool = False

        @property
        def _current_interval(self) -> int:
            return self._backoff_interval if self._consecutive_failures > 0 else self._poll_interval

        async def run(self, data_callback) -> None:
            """Main poll loop — runs until cancelled. (D-01: independent per device)"""
            next_poll = time.monotonic()
            while True:
                next_poll += self._current_interval
                now = time.monotonic()
                if next_poll > now:
                    await asyncio.sleep(next_poll - now)
                else:
                    # Overrun — reset anchor to now (D-07: skip missed, schedule from current time)
                    self.logger.debug(f"Poll overrun by {now - next_poll:.1f}s — scheduling immediately")
                    next_poll = now
                await self._poll_cycle(data_callback)

        async def _poll_cycle(self, data_callback) -> None:
            # Anti-overlap guard (D-07)
            if self._polling_in_progress:
                self.logger.debug("Poll overlap prevented — skipping this cycle")
                return
            self._polling_in_progress = True
            try:
                runtime = self._cycle_count * self._poll_interval  # D-09: elapsed-time anchored
                requests = self.parser.schedule_requests(runtime)
                if not requests:
                    return

                # Execute each scheduled request batch (replicates HA's execute_bulk pattern)
                responses = {}
                for req in requests:
                    code = req[REQUEST_CODE]
                    start = req[REQUEST_START]
                    count = req[REQUEST_COUNT]
                    result = await self.client.execute(code, start, count=count)
                    responses[(code, start)] = result

                parsed = self.parser.process(responses)
                self._handle_success()
                await data_callback(self.config.name, parsed)
            except ValueError as e:
                # D-10: invalidate_all triggered — drop entire cycle
                self.health.report_invalid_data(self.logger, str(e))
            except (TimeoutError, ConnectionError, OSError) as e:
                self._handle_failure(e)
            except asyncio.CancelledError:
                raise  # Let CancelledError propagate for clean shutdown
            except Exception as e:
                self.logger.error(f"Unexpected error during poll: {e!r}")
                self._handle_failure(e)
            finally:
                self._cycle_count += 1
                self._polling_in_progress = False

        def _handle_success(self) -> None:
            self.health.report_success(self.logger)
            self._consecutive_failures = 0
            self._backoff_interval = self._poll_interval

        def _handle_failure(self, error: Exception) -> None:
            self.health.report_failure(self.logger, error)
            self._consecutive_failures += 1
            self._backoff_interval = min(
                self._poll_interval * (BACKOFF_FACTOR ** self._consecutive_failures),
                MAX_BACKOFF,
            )
    ```

    **Sleep detection (D-04) — `_detect_solar` implementation detail:**
    After `parser.init()`, the items list has been preprocessed. Each item inherits its group's properties via `preprocess_descriptions` in common.py. However, the group name itself isn't directly stored on items. The simplest reliable approach: scan the parser's `_items` list for any item whose key or name contains "pv" (case-insensitive), since PV-related items like "PV1 Voltage", "PV Power" etc. are named consistently across all inverter profiles. Alternatively, re-load the raw profile YAML and check group names. Use the simpler approach — check item names:
    ```python
    def _detect_solar(parser: ParameterParser) -> bool:
        return any("pv" in item.get("name", "").lower() for item in parser._items if item.get("registers"))
    ```
    This catches "PV1 Voltage", "PV2 Current", "PV Power", etc. in inverter profiles, and correctly returns False for meter profiles which have no PV items.
