```python
# HA's homeassistant.util.slugify(text, separator="_") uses python-slugify under the hood.
# Standalone replacement:
from slugify import slugify as _slugify

def slugify(*items, separator: str = "_"):
    text = separator.join(filter(None, items))
    return _slugify(text, separator=separator)
```
